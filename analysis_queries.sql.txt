-- 1. Total Patients
SELECT COUNT(*) AS total_patients FROM Patients;

-- 2. Most Common Disease
SELECT disease, COUNT(*) AS total_cases
FROM Visits
GROUP BY disease
ORDER BY total_cases DESC
LIMIT 1;

-- 3. Visits Trend by Date
SELECT visit_date, COUNT(*) AS daily_visits
FROM Visits
GROUP BY visit_date
ORDER BY visit_date;

-- 4. Revenue by Doctor
SELECT doctor, SUM(bill_amount) AS total_income
FROM Visits
GROUP BY doctor
ORDER BY total_income DESC;

-- 5. Gender Distribution
SELECT gender, COUNT(*) AS total
FROM Patients
GROUP BY gender;

-- 6. City-wise Patients
SELECT city, COUNT(*) AS total_patients
FROM Patients
GROUP BY city
ORDER BY total_patients DESC;

-- 7. Most Used Medicine
SELECT medicine_name, COUNT(*) AS usage_count
FROM Medicines
GROUP BY medicine_name
ORDER BY usage_count DESC;
